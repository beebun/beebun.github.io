class UserStreakHistoryView(LoggingMixin, viewsets.ReadOnlyModelViewSet): 

        queryset = UserStreakHistory.objects.all().order_by('-created_at') 

        serializer_class = UserStreakHistorySerializer 

        filter_backends = (filters.DjangoFilterBackend,) 

        filter_fields = ('user', 'date') 

        paginate_by = 30 

 

        @list_route() 

        def logs(self, request): 

                daily_quest_templates_mapping = {} 

                dates = [] 

                start_date = create_datetime('2015-11-02 00:00:01', 'Asia/Bangkok').date() 

                end_date = get_today_date('Asia/Bangkok') 

                while start_date <= end_date: 

                        dates.append(start_date) 

                        start_date += timedelta(days=1) 

                student_profiles = StudentProfile.objects.all() 

                for student_profile in student_profiles: 

                        streak_data = {'streak': 0, 'streak_debug': 0} 

                        student = student_profile.get_user() 

                        classroom = student_profile.get_classroom() 

                        classroom_id = str(classroom.id) 

                        if classroom_id not in daily_quest_templates_mapping: 

                                daily_quest_templates_mapping[classroom_id] = {} 

                        streaks = UserStreakHistory.objects.filter( 

                                Q(user=student), 

                                Q(date__gte=dates[0]), 

                                Q(date__lte=dates[-1]) 

                        ).values_list('date', flat=True) 

                        for date in dates: 

                                date_index = str(date) 

                                streak = date in streaks 

                                if date_index not in daily_quest_templates_mapping[classroom_id]: 

                                        exists = DailyQuestTemplate.objects.filter( 

                                                classroom=classroom, 

                                                assigned_date=date 

                                        ).exists() 

                                        daily_quest_templates_mapping[classroom_id][date_index] = exists 

                                if not daily_quest_templates_mapping[classroom_id][date_index]: 

                                        streak_data['streak_debug'] = 0 

                                else: 

                                        if streak: 

                                                streak_obj = UserStreakHistory.objects.get(user=student, date=date) 

                                                streak_data['streak'] += 1 

                                                streak_data['streak_debug'] += 1 

                                                streak_obj.streak = streak_data['streak'] 

                                                streak_obj.streak_debug = streak_data['streak_debug'] 

                                                streak_obj.save() 

                                        else: 

                                                streak_data['streak'] = 0 

                                                streak_data['streak_debug'] = 0 

                return Response(success_response('Update Streak Successfully'))
