                        Q(daily_quest__course=template.get_course()), 

                        Q(daily_quest__assigned_date=template.get_assigned_date()), 

                        Q(question=question), 

                        Q(is_completed=True) 

                ) 

                for each in question_histories: 

                        if each.get_is_correct(): 

                                json['correct'] += 1 

                        else: 

                                json['incorrect'] += 1 

                if json['correct'] + json['incorrect'] > 0: 

                        json['correct_percent'] = float(json['correct']/float(json['correct']+json['incorrect']))*100
